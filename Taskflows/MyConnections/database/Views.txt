CREATE OR REPLACE FORCE VIEW "WCS_WEBCENTER"."CONNECTIONS_SEARCH_V" ("USERNAME", "CONNECTION_ID", "CONNECTION_USERNAME", "CONNECTION_DISPLAY_NAME", "CONNECTION_EMAIL", "CREATION_DATE")
AS
  SELECT AD.ACTOR_NAME USERNAME,
    AD2.ACTOR_ID CONNECTION_ID,
    AD2.ACTOR_NAME CONNECTION_USERNAME,
    AD2.DISPLAY_NAME CONNECTION_DISPLAY_NAME,
    AD2.EMAIL CONNECTION_EMAIL,
    PC.CREATION_DATE CREATION_DATE
  FROM WCS_WEBCENTER.WC_PEOPLE_CONN_CONNECTION PC,
    WCS_WEBCENTER.WC_AS_ACTOR_DETAIL AD,
    WCS_WEBCENTER.WC_AS_ACTOR_DETAIL AD2
  WHERE (PC.OWNER_ID =AD.ACTOR_ID
  AND PC.CONNECTEE_ID=AD2.ACTOR_ID)
  OR (PC.CONNECTEE_ID=AD.ACTOR_ID
  AND PC.OWNER_ID    =AD2.ACTOR_ID);
  
  
  
  
  SELECT USERNAME,
    CONNECTION_ID,
    CONNECTION_USERNAME,
    CONNECTION_DISPLAY_NAME,
    CONNECTION_EMAIL,
    CREATION_DATE
  FROM WCS_WEBCENTER.CONNECTIONS_SEARCH_V
WHERE upper(USERNAME)=upper('WEBLOGICADMIN');
